    <script type="module">

        import { collection, query, onSnapshot, addDoc, updateDoc, doc, deleteDoc, orderBy, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        let allJobs = [];


        // 加載資料

        window.loadJobs = () => {

            if (!window.currentUser) return;

            const q = query(

                collection(window.db, 'artifacts', window.appId, 'users', window.currentUser.uid, 'jobs'),

                orderBy('createdAt', 'desc')

            );

            

            onSnapshot(q, (snapshot) => {

                allJobs = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

                renderJobs(allJobs);

                updateStats(allJobs);

            }, (error) => {

                console.error("Firestore error:", error);

            });

        };


        // 渲染表格

        window.renderJobs = (jobs) => {

            const table = document.getElementById('job-table-body');

            if (jobs.length === 0) {

                table.innerHTML = `<tr><td colspan="5" class="p-20 text-center text-slate-400">目前沒有追蹤職缺。使用 AI 分析來新增第一個吧！</td></tr>`;

                return;

            }


            table.innerHTML = '';

            jobs.forEach(job => {

                const row = document.createElement('tr');

                row.className = "hover:bg-indigo-50/30 transition-all group";

                row.innerHTML = `

                    <td class="px-6 py-5">

                        <div class="font-bold text-slate-800 leading-tight">${job.title || '職缺'}</div>

                        <div class="text-xs text-slate-400 mt-1">${job.company || '公司'}</div>

                    </td>

                    <td class="px-6 py-5 text-xs font-semibold text-slate-500">${job.type || 'Intern'}</td>

                    <td class="px-6 py-5">

                        <select onchange="updateJobStatus('${job.id}', this.value)" class="status-badge ${getStatusClass(job.status)}">

                            <option value="未投遞" ${job.status === '未投遞' ? 'selected' : ''}>未投遞</option>

                            <option value="已投遞" ${job.status === '已投遞' ? 'selected' : ''}>已投遞</option>

                            <option value="面試中" ${job.status === '面試中' ? 'selected' : ''}>面試中</option>

                            <option value="拒信" ${job.status === '拒信' ? 'selected' : ''}>拒信</option>

                            <option value="錄取" ${job.status === '錄取' ? 'selected' : ''}>錄取</option>

                        </select>

                    </td>

                    <td class="px-6 py-5 text-center">

                        <span class="px-3 py-1 rounded-lg bg-slate-100 text-sm font-black ${job.score >= 8 ? 'text-green-600' : (job.score >= 6 ? 'text-amber-600' : 'text-slate-400')}">

                            ${job.score || '0.0'}

                        </span>

                    </td>

                    <td class="px-6 py-5 text-right">

                        <button onclick="deleteJob('${job.id}')" class="p-2 text-slate-300 hover:text-red-500 transition-colors">

                            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path></svg>

                        </button>

                    </td>

                `;

                table.appendChild(row);

            });

        };


        window.updateJobStatus = async (id, newStatus) => {

            const jobRef = doc(window.db, 'artifacts', window.appId, 'users', window.currentUser.uid, 'jobs', id);

            await updateDoc(jobRef, { status: newStatus });

        };


        window.deleteJob = async (id) => {

            if (!confirm("確定要刪除這個職缺追蹤嗎？")) return;

            const jobRef = doc(window.db, 'artifacts', window.appId, 'users', window.currentUser.uid, 'jobs', id);

            await deleteDoc(jobRef);

        };


        window.addNewJobDirectly = async () => {

            const title = prompt("輸入職稱:");

            const company = prompt("輸入公司名稱:");

            if (!title || !company) return;

            

            await addDoc(collection(window.db, 'artifacts', window.appId, 'users', window.currentUser.uid, 'jobs'), {

                title, company, status: '未投遞', type: 'Intern', score: 7.0, createdAt: serverTimestamp()

            });

        };


        window.filterJobs = () => {

            const term = document.getElementById('search-input').value.toLowerCase();

            const filtered = allJobs.filter(j => 

                (j.title?.toLowerCase().includes(term)) || 

                (j.company?.toLowerCase().includes(term))

            );

            renderJobs(filtered);

        };


        function getStatusClass(status) {

            if (status === '已投遞') return 'status-applied shadow-sm';

            if (status === '面試中') return 'status-interview shadow-sm';

            if (status === '拒信') return 'status-rejected shadow-sm';

            if (status === '錄取') return 'status-offer shadow-sm';

            return 'status-not-applied';

        }


        function updateStats(jobs) {

            const applied = jobs.filter(j => j.status !== '未投遞').length;

            const interviews = jobs.filter(j => j.status === '面試中').length;

            const offers = jobs.filter(j => j.status === '錄取').length;

            const validScores = jobs.map(j => parseFloat(j.score)).filter(s => !isNaN(s));

            const avgScore = validScores.length > 0 ? (validScores.reduce((a, b) => a + b, 0) / validScores.length).toFixed(1) : "0.0";


            document.getElementById('stat-applied-ratio').innerText = `${applied} / ${jobs.length}`;

            document.getElementById('stat-interview').innerText = interviews;

            document.getElementById('stat-offers').innerText = offers;

            document.getElementById('stat-avg-score').innerText = avgScore;

        }

    <script>

        // AI 邏輯與介面控制

        const apiKey = ""; // 自動注入


        function toggleAIPanel() {

            const panel = document.getElementById('ai-panel');

        }


        async function analyzeJD() {

            const jd = document.getElementById('jd-input').value;

            if (!jd) return;


            const btn = document.getElementById('analyze-btn');

            const loader = document.getElementById('ai-loader');

            const resultDiv = document.getElementById('ai-result');


            btn.disabled = true;

            loader.classList.remove('hidden');

            resultDiv.innerHTML = '<div class="text-center py-10 animate-pulse text-indigo-600 font-bold">Gemini 正在精算匹配機率...</div>';


            const candidateBackground = `Pin-Chu (Chloe) Yin, MSBA @ BU. 核心優勢: Python/SQL, Amazon 1.8M 資料分析, 電商創業主, Garmin HR 實習.`;

            const prompt = `你是頂級求職顧問。請根據以下候選人背景和 JD，進行適配度分析。

            候選人: ${candidateBackground}

            JD: ${jd}

            請嚴格以 JSON 格式回傳: {"score": 數字0-10, "title": "職稱", "company": "公司", "type": "Intern或Full-time", "reason": "一句話總結匹配原因", "missing": "缺少的技能"}

            不要有額外文字。`;


            try {

                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{ parts: [{ text: prompt }] }],

                        generationConfig: { responseMimeType: "application/json" }

                    })

                });

                const data = await response.json();

                const res = JSON.parse(data.candidates[0].content.parts[0].text);

                

                resultDiv.innerHTML = `

                    <div class="space-y-4">

                        <div class="flex justify-between items-end">

                            <div class="font-black text-slate-800 text-xl">${res.company}</div>
